; Translating _samples/3_conditionals/gcd.py
; Branching to top level (tl) instructions
		BR tl
; Allocating Global (static) memory
a:       	.BLOCK 2
b:       	.BLOCK 2
; Top Level instructions
tl:      	NOP1
		DECI a,d
		DECI b,d
test_0:  	LDWA a,d
		CPWA b,d
		BREQ end_l_0
cond_1:  	LDWA a,d
		CPWA b,d
		BRLE else_1
if_1:    	NOP1
		LDWA a,d
		SUBA b,d
		STWA a,d
		BR test_0
		BR end_if_1
else_1:  	NOP1
		LDWA b,d
		SUBA a,d
		STWA b,d
end_if_1:	NOP1
		BR test_0
end_l_0: 	NOP1
		DECO a,d
		.END
